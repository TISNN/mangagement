  private maxConcurrent = 3;
  private running = 0;
  private queue: Array<() => Promise<any>> = [];

  async add(task: () => Promise<any>): Promise<any> {
    return new Promise((resolve, reject) => {
      this.queue.push(async () => {
        try {
          const result = await task();
          resolve(result);
        } catch (error) {
          reject(error);
        } finally {
          this.running--;
          this.processQueue();

  }
}
